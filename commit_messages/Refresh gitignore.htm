A watchdog csomag segítségével figyelheted a fájlokat és könyvtárakat:

Telepítés:

bash

Kód egy mappa figyelésére:

python
from watchdog.observers import Observer
from watchdog.events import FileSystemEventHandler
import time

class Watcher:
    DIRECTORY_TO_WATCH = "D:/RenoMate/renomate-backend"

    def __init__(self):
        self.event_handler = MyHandler()
        self.observer = Observer()

    def run(self):
        self.observer.schedule(self.event_handler, self.DIRECTORY_TO_WATCH, recursive=True)
        self.observer.start()
        try:
            while True:
                time.sleep(5)
        except KeyboardInterrupt:
            self.observer.stop()
        self.observer.join()

class MyHandler(FileSystemEventHandler):
    def on_modified(self, event):
        if not event.is_directory:
            print(f"Fájl módosítva: {event.src_path}")

    def on_created(self, event):
        print(f"Új fájl vagy mappa jött létre: {event.src_path}")

    def on_deleted(self, event):
        print(f"Fájl törölve: {event.src_path}")

if __name__ == "__main__":
    watcher = Watcher()
    watcher.run()

Ha minden mentéskor automatikusan pusholni akarod a kódot, akkor egy Python script is figyelheti a változásokat.

📌 4.1 Telepítsd a Watchdog csomagot

📌 4.2 Hozz létre egy automatikus Git push scriptet

Hozz létre egy fájlt auto_git_push.py néven a D:/RenoMate/renomate-backend mappában, és másold be ezt a kódot:

python
import time
import subprocess
from watchdog.observers import Observer
from watchdog.events import FileSystemEventHandler

class GitAutoPushHandler(FileSystemEventHandler):
    def on_modified(self, event):
        if event.is_directory:
            return
        print(f"Módosított fájl: {event.src_path}, automatikus push...")
        subprocess.run(["git", "add", "."], check=True)
        subprocess.run(["git", "commit", "-m", "Automatikus frissítés"], check=True)
        subprocess.run(["git", "push", "origin", "main"], check=True)

if __name__ == "__main__":
    path = "D:/RenoMate/renomate-backend"
    event_handler = GitAutoPushHandler()
    observer = Observer()
    observer.schedule(event_handler, path, recursive=True)
    observer.start()
    try:
        while True:
            time.sleep(5)
    except KeyboardInterrupt:
        observer.stop()
    observer.join()
